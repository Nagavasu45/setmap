Q1. "Problem: You are given a string (STR) of length N, consisting of only the lower case English alphabet. Your task is to remove all the duplicate occurrences of characters in the string. Input: abcadeecfb Output: abcdef" 

    

        function duplicates(N){
            let res=new Set();    
            for (let i= 0; i< N.length; i++){
                res.add(N[i])
            }
        return Array.from(res).join('');
    }
console.log(duplicates("abcadeecfb"))

Explaination:Here we have to solve the remove the duplicate lettes from the string..so i have used set() method for filtering like duplicates values not allow to store in set list...then i have pushed one by one letter to set() then we get set list but i have to convert the set list into string format soo i used Array.from(set_variable).join() then we get proper output

Q2. "Problem: You are given a string (STR) of length N, you have to print the count of all alphabet.(using maps) Input: abcadeecfb Output: a=2 b=2 c=2 d=1 e=2 f=1" 

        function countings(N){
            let res= new Map()
            for(let char of N){
                if(res.has(char)){
                    res.set(char,res.get(char)+1)
                }
                else{
                    res.set(char,1)
                }
            }
            let output=Array.from(res.keys())
            for(let key of output){
                console.log(`${key} = `+ res.get(key))
            }
        }
        countings("abcdabcd")

Explaination:here we have to print the count of all alphabet using map() and get char one by one through using for loop and in map() we have set and get methods set for assign the key, value and get for getting value of key
